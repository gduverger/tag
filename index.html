Unobtrusive scripting

<!-- Loads jQuery (through Google CDN) -->
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.4.2/jquery.min.js"></script>

<!-- Loads the Tag script -->
<script src="http://simplicityengineer.com/tag/jquery.tag.js"></script>

<!-- Loads the Tag style sheet (or make your own!) -->
<link rel="stylesheet" href="http://simplicityengineer.com/tag/tag.css">

<!-- Generate your list of existing tags -->
<ul class="tags">
  <li>Aggregators</li>
  <li>Wikis</li>
</ul>

<!-- Binds tag editing; fallback in case JavaScript is disabled -->
<a href="editTags.html" class="edit-tags">Edit tags</a>

<script>
  $(document).ready(function() {
    // Binds the tag container (has to be the first parent)
    $(".tags").tag({
      // Defines how new tags are formed
      tagTemplate: "<li>",
      // Defines what triggers the edit mode
      editTagsSelector: ".edit-tags",
      // Called when a tag is saved
      saveTagCallback: function( tag ) {
        // AJAX call to update the tag on your server
        return true;
      },
      // Called when a tag is removed
      removeTagCallback: function( tag ) {
        // AJAX call to remove the tag from your server
        return true;
      }
    });
  });
</script>
